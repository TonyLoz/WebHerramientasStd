CREATE TABLE `tb_certificador` (
  `ID_CERT` int(11) NOT NULL AUTO_INCREMENT,
  `NOMBRE` varchar(100) NOT NULL,
  `SELECCIONADO` int(4) DEFAULT '0',
  PRIMARY KEY (`ID_CERT`),
  UNIQUE KEY `NOMBRE` (`NOMBRE`)
) ENGINE=InnoDB AUTO_INCREMENT=6 DEFAULT CHARSET=utf8;

CREATE TABLE `tb_bitacora` (
  `IDB` int(11) NOT NULL AUTO_INCREMENT,
  `USUARIO` varchar(200),
  `FECHA` datetime,
  `OPERACION` varchar(80),
  `COMENTARIOS` varchar(200),
  `ARCHIVO` varchar(100),
  PRIMARY KEY (`IDB`)
) ENGINE=InnoDB AUTO_INCREMENT=6 DEFAULT CHARSET=utf8;
